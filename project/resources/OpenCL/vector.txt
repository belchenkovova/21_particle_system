typedef struct
{
	float			x;
	float			y;
	float			z;
}					vector;

static vector		vector_add(vector a, vector b)
{
	return ((vector){a.x + b.x, a.y + b.y, a.z + b.z});
}

static vector		vector_subtract(vector a, vector b)
{
	return ((vector){a.x - b.x, a.y - b.y, a.z - b.z});
}

static vector		vector_multiply(vector a, float value)
{
	return ((vector){a.x * value, a.y * value, a.z * value});
}

static vector		vector_divide(vector v, float value)
{
	return (vector_multiply(v, 1.f / value));
}

static float		vector_length_squared(vector v)
{
	return (v.x * v.x + v.y * v.y + v.z * v.z);
}

static float		vector_length(vector v)
{
	return (sqrt(vector_length_squared(v)));
}

static vector		vector_normalize(vector v)
{
	return (vector_divide(v, vector_length(v)));
}
