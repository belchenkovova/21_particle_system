static void			reset_target(
					global vector *position,
					global vector *velocity,
					global vector *acceleration,
					int id)
{
	position[id].x = 0;
    position[id].y = 0;
    position[id].z = 0;

	velocity[id].x = 0;
	velocity[id].y = 0;
	velocity[id].z = 0;

	acceleration[id].x = 0;
	acceleration[id].y = 0;
	acceleration[id].z = 0;
}

kernel void			reset(
					global vector *position,
					global vector *velocity,
					global vector *acceleration)
{
	reset_target(position, velocity, acceleration, get_global_id(0));
}
